<script>
    function initpagination() {
        jQuery("#buttonenvoie").click(function(event){
            jQuery('#offset').val(0);
            jQuery('#formsession').val('');
        });
        jQuery("ul.pagination li a").click(function(event){
            var href = jQuery(this).attr("href").split("/");
            offset=0;
            if(href[1])
            {
                offset=href[1];
            }
    	    jQuery('#offset').val(offset);
            jQuery("#frm-annuaire").attr("action","https://www.algerietelecom.dz/fr/annuaire/search/" + offset);
            jQuery("#frm-annuaire").submit();
            event.preventDefault();
        });
    }
</script>
<script type="text/javascript">
    function changerwilaya(){
        var wilaya = jQuery('#wilaya').val();
        jQuery('#block_commune').html('Chargement en cours ...');
        jQuery.ajax({type:"POST",data:({}), url:"https://www.algerietelecom.dz/fr/annuaire/commune/" + wilaya,
            success: function(msg){
                jQuery('#block_commune').html(msg);
            },
            complete : function(){
            },
            error:function(XMLHttpRequest,textStatus, errorThrown){
            }
        });
        return false;
    }

    jQuery("#frm-annuaire").submit(function(){
        jQuery('#buttonenvoie i').show();
        jQuery('#buttonenvoie span').hide();
        jQuery('#alert').html('');
        jQuery('#alertaucun').html('');
        jQuery('#alert').removeClass('alert');
        jQuery('#alert').removeClass('alert-success');
        jQuery('#alert').removeClass('alert-danger');
        jQuery('.column .text-danger').html('');
        var offset = jQuery('#offset').val();
        jQuery.ajax({type:"POST",data:jQuery(this).serialize(), url:"https://www.algerietelecom.dz/fr/annuaire/search/" + offset,
            dataType: 'json',
            success: function(msg)
            {
                grecaptcha.reset();
                if(msg.resultat == 'ok'){
                    jQuery('#annuairecontent').html(msg.content);
                    initpagination();
                    jQuery('#formsession').val(msg.formsession);
                }else{
                    jQuery('#formsession').val('');
                    jQuery('#alert').html('Erreur dans vos informations de contact');
                    jQuery('#alert').addClass('alert alert-danger');
                    content = msg.content;
                    for( id in content){
                        jQuery('#frm-annuaire #block_'+id).addClass('has-error');
                        jQuery('#frm-annuaire #block_'+id+' .text-danger').html(content[id]);
                    }
                    jQuery(".has-error").click(function(){
                        jQuery('#frm-annuaire #'+jQuery(this).attr("id")+' .text-danger').html('');
                        jQuery(this).removeClass("has-error");
                    });
                    window.location.href = '#alert';
                }
            },
            complete : function(){
                jQuery('#buttonenvoie span').show();
                jQuery('#buttonenvoie i').hide();
            },
            error:function(XMLHttpRequest,textStatus, errorThrown){
                alert("(0086) Erreur de chargement..."+XMLHttpRequest.responseText);
                jQuery('#alert').html(XMLHttpRequest.responseText);
            }
        });
        return false;
    });
</script>
